✅ 3.3　文字列から数値への変換


👉  なぜ変換が必要なのか
-----------------------
console.log('5000' + 3);   // → '50003'（数値の足し算ではなく文字列の結合！）
console.log('5000' - 3);   // → 4997（引き算は自動で数値変換される）
console.log('5000' * 3);   // → 15000（掛け算も自動変換）
-----------------------


👉  3つの変換方法
-----------------------
const input = '5000';
 
// 方法1: Number() — 汎用。小数も整数も変換できる（推奨）
const n1 = Number(input);    // → 5000
 
// 方法2: parseInt() — 整数のみ。小数点以下を切り捨てる
const n2 = parseInt(input);  // → 5000
const n3 = parseInt('3.9');  // → 3（切り捨て）
 
// 方法3: parseFloat() — 小数点ありの数値に使う
const n4 = parseFloat('3.14'); // → 3.14
-----------------------


👉  NaN（Not a Number）に注意する
-----------------------
Number('abc');     // → NaN
Number('');        // → 0（空文字は0になる！）
Number('  ');      // → 0（スペースのみも0）
parseInt('abc');   // → NaN
 
// NaN のチェックは isNaN() を使う
console.log(isNaN(Number('abc')));  // → true
console.log(isNaN(5000));            // → false
-----------------------


👉  app.js に変換処理を追加する

👉  丸め処理の方法
-----------------------
const raw = 1666.6666;
 
// Math.round() — 四捨五入（最もよく使う）
Math.round(raw);        // → 1667
 
// Math.ceil() — 切り上げ（端数は多い側が払う場合に使う）
Math.ceil(raw);         // → 1667
 
// Math.floor() — 切り捨て
Math.floor(raw);        // → 1666
 
// toFixed(n) — 小数点以下n桁で文字列として返す
raw.toFixed(0);         // → '1667'（文字列！）
raw.toFixed(2);         // → '1666.67'

✅ 3.6　エラーハンドリングとバリデーション

👉 バリデーション関数を作る
-----------------------
/**
 * 入力値を検証する
 * @param {number} total  - 合計金額
 * @param {number} people - 人数
 * @returns {string|null} - エラーメッセージ、問題なければ null
 */
function validate(total, people) {
  if (totalInput.value === '' || peopleInput.value === '') {
    return '合計金額と人数を入力してください';
  }
  if (isNaN(total) || isNaN(people)) {
    return '数字を入力してください';
  }
  if (total < 0) {
    return '合計金額は0以上を入力してください';
  }
  if (people <= 0) {
    return '人数は1以上を入力してください';
  }
  return null;  // エラーなし
}
-----------------------


👉  バリデーションをイベントリスナーに組み込む
-----------------------
calcBtn.addEventListener('click', () => {
  const total  = Number(totalInput.value);
  const people = Number(peopleInput.value);
 
  // バリデーション
  const errorText = validate(total, people);
  if (errorText !== null) {
    errorMsg.textContent = errorText;
    resultAmount.textContent = '——';  // 前の結果をリセット
    resultTax.textContent    = '';
    resultSection.classList.remove('active');
    return;
  }
 
  // バリデーション通過後の処理
  errorMsg.textContent = '';
 
  const perPerson    = Math.round(total / people);
  const perPersonTax = Math.round((total * (1 + TAX_RATE)) / people);
 
  resultAmount.textContent = perPerson.toLocaleString() + '円';
  resultTax.textContent    = `税込 ${perPersonTax.toLocaleString()}円`;
  resultSection.classList.add('active');
});

✅ 3.7　UIのブラッシュアップ


👉  追加ボタンのイベントリスナー
-----------------------
const expenseName     = document.querySelector('#expenseName');
const expenseAmount   = document.querySelector('#expenseAmount');
const expenseCategory = document.querySelector('#expenseCategory');
const addBtn          = document.querySelector('#addBtn');
const expenseList     = document.querySelector('#expenseList');
 
// 支出データを配列で管理
let expenses = [];
 
addBtn.addEventListener('click', () => {
  const name     = expenseName.value.trim();
  const amount   = Number(expenseAmount.value);
  const category = expenseCategory.value;
 
  // 入力チェック
  if (name === '' || isNaN(amount) || amount <= 0) {
    alert('項目名と金額を正しく入力してください');
    return;
  }
 
  // 支出オブジェクトを作成して配列に追加
  const expense = {
    id:       Date.now(),  // ミリ秒タイムスタンプで一意IDを生成
    name,
    amount,
    category,
  };
 
  expenses.push(expense);
  renderList();           // リストを再描画
 
  // 入力フォームをリセット
  expenseName.value   = '';
  expenseAmount.value = '';
});
-----------------------


👉  リストを画面に描画する関数
-----------------------
function renderList() {
  // 一度全部消してから描画し直す
  expenseList.innerHTML = '';
 
  expenses.forEach((expense) => {
    const li = document.createElement('li');
    li.className = 'expense-item';
    li.dataset.id = expense.id;  // 削除時に使うIDをdata属性に保存
 
    li.innerHTML = `
      <span class="category-badge">${expense.category}</span>
      <span class="expense-name">${expense.name}</span>
      <span class="expense-amount">${expense.amount.toLocaleString()}円</span>
      <button class="delete-btn" data-id="${expense.id}">削除</button>
    `;
 
    expenseList.appendChild(li);
  });
 
  // 合計もあわせて更新
  updateTotal();
}
-----------------------


👉  削除機能（イベント委任パターン）
-----------------------
expenseList.addEventListener('click', (event) => {
  // クリックされたのが削除ボタンかどうかを確認
  if (!event.target.classList.contains('delete-btn')) return;
 
  const id = Number(event.target.dataset.id);
 
  // 該当IDを除いた新しい配列を作る（元の配列は変更しない）
  expenses = expenses.filter(e => e.id !== id);
  renderList();
});
-----------------------


👉  支出合計を自動計算する
-----------------------
function updateTotal() {
  const total = expenses.reduce((acc, e) => acc + e.amount, 0);
 
  // 合計金額フィールドに自動入力
  totalInput.value = total;
 
  // 人数が入力されていれば計算も自動更新
  const people = Number(peopleInput.value);
  if (people > 0) {
    const perPerson    = Math.round(total / people);
    const perPersonTax = Math.round((total * (1 + TAX_RATE)) / people);
    resultAmount.textContent = perPerson.toLocaleString() + '円';
    resultTax.textContent    = `税込 ${perPersonTax.toLocaleString()}円`;
    resultSection.classList.add('active');
  }
}
